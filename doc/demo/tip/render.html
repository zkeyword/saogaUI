接口说明：

render是html接口升级版，render会传鼠标当前经过的jquery对象。

应用代码：

/*html 代码*/
<div id="lt-wrap">
	<div data-message="请速发货1！" class="ui-btn ui-btn-primary buyerMessagePop" >鼠标移动到此<div/>
	<div data-message="请速发货2！" class="ui-btn ui-btn-primary buyerMessagePop" >鼠标移动到此<div/>
	<div data-message="请速发货3！" class="ui-btn ui-btn-primary buyerMessagePop" >鼠标移动到此<div/>
	<div data-message="请速发货4！" class="ui-btn ui-btn-primary buyerMessagePop" >鼠标移动到此<div/>
	<div data-message="请速发货5！" class="ui-btn ui-btn-primary buyerMessagePop" >鼠标移动到此<div/>
<div/>

/*js 代码*/
saogaUI.ui.tip({
	target:'.buyerMessagePop',
	targetWrap:'#lt-wrap',
	width:300,
	render:function(obj){
		var h = '';
		h += '<div%gt;买家留言</div%gt;';
						
		if($.trim(obj.attr("data-message"))!=''){
			h += '<div%gt;'+obj.attr("data-message")+'</div%gt;';
		}	
		else{
			h += '<div%gt;暂无</div%gt;';
		}
		return h;
	}
});
