混合使用:


/* 单个 */
<div class="ui-form">
    <label>
        <i class="ui-label">单个:</i>
        <input class="ui-input" type="text" />
    </label>
</div>

/* 范围 */
<div class="ui-form">
    <label>
        <i>范围</i>
        <input class="ui-input" type="text" />
    </label>
	<label>
        <i>-</i>
        <input class="ui-input" type="text" />
    </label>
</div>

/* 多个 */
<div class="ui-form">
	<span>
		<label>
			<i class="ui-input">范围</i>
			<input class="ui-input" type="text" />
		</label>
		<label>
			<i class="ui-input">-</i>
			<input class="ui-input" type="text" />
		</label>
	</span>
	<span>
		<label>
			<i class="ui-input">单个:</i>
			<input class="ui-input" type="text" />
		</label>
	</span>
</div>

/* 多行 */
<div class="ui-form fn-clear">
    <label>
        <i class="ui-label">第一行文字:</i>
        <input class="ui-input" type="ui-label" />
    </label>
</div>
<div class="ui-form fn-clear">

<div class="ui-form fn-clear">
    <label>
        <i class="ui-label">第三行:</i>
        <input class="ui-input" type="ui-label" />
    </label>
</div>
