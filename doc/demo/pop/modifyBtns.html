接口说明：

修改按钮的API是modifyBtns，只要传入和原来的弹窗框的按钮数据格式一致即可。

应用代码：

var html = '<div>这个是一个pop弹出框</div>';

var pop = saogaUI.ui.pop({
	title:'弹窗的标题',
	height:240,
	width:320,
	html:html,
	btns:[
		{
			text:'保存修改',
			closePop:false,
			onclick:function(id){
				alert('保存修改')
				pop.close(id);
			}
		},
		{
			text:'取消',
			onclick:function(){
	
			}
		}
	]
});

pop.modifyBtns([
	{
		text:'保存',
		closePop:false,
		onclick:function(id){
			alert('保存')
			pop.close(id);
		}
	},
	{
		text:'关闭',
		onclick:function(){

		}
	}
]);
