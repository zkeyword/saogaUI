对话框
alert：点击打开

						$('#dialog_alert').click(function(){
							saogaUI.ui.dialog.alert({
								title:'标题',
								text:'内容'
							});
						});

confirm：点击打开

						$('#dialog_confirm').click(function(){
							saogaUI.ui.dialog.confirm({
								title:'标题',
								text:'内容'
							});
						});

error：点击打开

						$('#dialog_error').click(function(){
							saogaUI.ui.dialog.error({
								title:'标题',
								text:'内容'
							});
						});

prompt：点击打开

						$('#dialog_error').click(function(){
							saogaUI.ui.dialog.prompt({
								text:'测试测试',
								showTime:'5000' //5秒后关闭,
								endFn: function(){
									alert('sssss');
								}
							});
						});
